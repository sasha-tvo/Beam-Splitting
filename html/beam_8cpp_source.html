7 // Parameters: e - old basis,

11 // | -sin(angle) cos(angle) |
12 void Rot(const Point3D& e, const Point3D& E, const Point3D& v, matrixC &m)
13 {
14  const double Eps = 1e2*DBL_EPSILON; // acceptable precision
15  double cs = E*e;
16  if(fabs(1.0-cs) < Eps) return;
17  if(fabs(1.0+cs) < Eps) {
18  m[0][0] = -m[0][0]; m[0][1] = -m[0][1];
19  m[1][0] = -m[1][0]; m[1][1] = -m[1][1];
20  return;
21  }
22  #ifdef _DEBUG
23  if(fabs(cs) > 1.0+DBL_EPSILON) throw " Rot: Error!";
24  #endif
25  double angl = acos(cs);
26  Point3D k = e%E;
27  k /= length(k);
28  if(norm(k+v) <= 0.5) angl = -angl;
29  double sn = sin(angl); // the rotation of matrix "m"
30  complex b00 = m[0][0]*cs+m[1][0]*sn, // first row of the result
31  b01 = m[0][1]*cs+m[1][1]*sn;
32  m[1][0] = m[1][0]*cs-m[0][0]*sn;
33  m[1][1] = m[1][1]*cs-m[0][1]*sn;
34  m[0][0] = b00; m[0][1] = b01; //
35 }
